Project: Finding Lane Lines on the Road¶

In this project, you will use the tools you learned about in the lesson to identify lane lines on the road. You can develop your pipeline on a series of individual images, and later apply the result to a video stream (really just a series of images). Check out the video clip "raw-lines-example.mp4" (also contained in this repository) to see what the output should look like after using the helper functions below.

Once you have a result that looks roughly like "raw-lines-example.mp4", you'll need to get creative and try to average and/or extrapolate the line segments you've detected to map out the full extent of the lane lines. You can see an example of the result you're going for in the video "P1_example.mp4". Ultimately, you would like to draw just one line for the left side of the lane, and one for the right.

import math

def grayscale(img):
    """Applies the Grayscale transform
    This will return an image with only one color channel
    but NOTE: to see the returned image as grayscale
    (assuming your grayscaled image is called 'gray')
    you should call plt.imshow(gray, cmap='gray')"""
    return cv2.cvtColor(img, cv2.COLOR_RGB2GRAY)
    # Or use BGR2GRAY if you read an image with cv2.imread()
    # return cv2.cvtColor(img, cv2.COLOR_BGR2GRAY)
    
def canny(img, low_threshold, high_threshold):
    """Applies the Canny transform"""
    return cv2.Canny(img, low_threshold, high_threshold)

def gaussian_blur(img, kernel_size):
    """Applies a Gaussian Noise kernel"""
    return cv2.GaussianBlur(img, (kernel_size, kernel_size), 0)

def region_of_interest(img, vertices):
    """
    Applies an image mask.
    
    Only keeps the region of the image defined by the polygon
    formed from `vertices`. The rest of the image is set to black.
    """
    #defining a blank mask to start with
    mask = np.zeros_like(img)   
    
    #defining a 3 channel or 1 channel color to fill the mask with depending on the input image
    if len(img.shape) > 2:
        channel_count = img.shape[2]  # i.e. 3 or 4 depending on your image
        ignore_mask_color = (255,) * channel_count
    else:
        ignore_mask_color = 255
        
    #filling pixels inside the polygon defined by "vertices" with the fill color    
    cv2.fillPoly(mask, vertices, ignore_mask_color)
    
    #returning the image only where mask pixels are nonzero
    masked_image = cv2.bitwise_and(img, mask)
    return masked_image

def draw_lines(img, lines, color=[255, 0, 0], thickness=15):
    """
    Draw boundaries in Red and path in Green.
    
    """
    left_lane = np.zeros((1,4), dtype=np.int) # Most dominant left lane
    right_lane = np.zeros((1,4), dtype=np.int)  # Most dominant right lane
    global last_known_good_left_line
    global last_known_good_right_line

    # Find the most dominant lines
    for line in lines:
        for x1, y1, x2, y2 in line:
            a = np.array((x1, y1))
            b = np.array((x2, y2))
            slope = (y2 - y1)/(x2 - x1)
            if(slope >= 0.5): # Right
                add_if_greater(right_lane, x1, y1, x2, y2, "right", img.shape[1])
            elif(slope <= -0.5): # Left
                add_if_greater(left_lane, x1, y1, x2, y2, "left", img.shape[1])

    # Use last known good lane if lane was not found in this frame ** This needs to reset after few failed frames
    if(found_lane(left_lane, right_lane)):
        last_known_good_left_line = left_lane
        last_known_good_right_line = right_lane
    else:
        left_lane = last_known_good_left_line
        right_lane = last_known_good_right_line

    # Draw Green polygon and Red lane lines using the left and right lane points
    poly = np.array( [[[left_lane[0][0], left_lane[0][1]],[left_lane[0][2], left_lane[0][3]],
             [right_lane[0][0], right_lane[0][1]],[right_lane[0][2], right_lane[0][3]]]], dtype=np.int32)
    cv2.fillPoly(img, poly, [0, 80, 0])
    cv2.line(img, (left_lane[0][0], left_lane[0][1]), (left_lane[0][2], left_lane[0][3]), color, thickness)
    cv2.line(img, (right_lane[0][0], right_lane[0][1]), (right_lane[0][2], right_lane[0][3]), color, thickness)

def found_lane(left_lane, right_lane):
    """If points are zeros then lane was not found"""
    if((left_lane[0][0] == 0 and left_lane[0][1] == 0) or
       (left_lane[0][2] == 0 and left_lane[0][3] == 0) or
       (right_lane[0][0] == 0 and right_lane[0][1] == 0) or
       (right_lane[0][2] == 0 and right_lane[0][3] == 0)):
        return False
    else:
        return True
      
def nonzero_mean(array):
    """Calculate mean without considering zero elements"""
    totals = np.zeros((4), dtype=np.int)
    counter = 0
    for element in array:
        if(element[0] + element[1] + element[2] + element[3] > 0):
            counter += 1
        for ii in range(0, 4):
            totals[ii] += element[ii]
    if (counter == 0):
        return totals
    for jj in range(0, 4):
        totals[jj] /= counter
    return totals

def add_if_greater(myArray, x1, y1, x2, y2, which, h):
    """Add line if it's most dominant"""
    a = np.array((x1, y1))
    b = np.array((x2, y2))
    global avg_left
    global avg_right
    global lane_index
    global moving_frame_avg
    for i in range(0, len(myArray)):
        if(np.linalg.norm(a - b) > 
           np.linalg.norm(
               np.array((myArray[i][0], myArray[i][1])) - 
               np.array(((myArray[i][2], myArray[i][3]))))):
            if(which == "right"):
                lane_index[1] += 1
                lane_index[1] %= moving_frame_avg
                avg_right[lane_index[1]] = [x1, y1, x2, y2]
                myArray[i] = nonzero_mean(avg_right)
            else: # left lane
                lane_index[0] += 1
                lane_index[0] %= moving_frame_avg
                avg_left[lane_index[0]] = [x1, y1, x2, y2]
                myArray[i] = nonzero_mean(avg_left)
    return myArray
            
def hough_lines(img, rho, theta, threshold, min_line_len, max_line_gap):
    """
    `img` should be the output of a Canny transform.
        
    Returns an image with hough lines drawn.
    """
    lines = cv2.HoughLinesP(img, rho, theta, threshold, np.array([]), minLineLength=min_line_len, maxLineGap=max_line_gap)
    line_img = np.zeros((img.shape[0], img.shape[1], 3), dtype=np.uint8)
    draw_lines(line_img, lines)
    return line_img

    `img` is the output of the hough_lines(), An image with lines drawn on it.
    Should be a blank image (all black) with lines drawn on it.
    
    `initial_img` should be the image before any processing.
    
    The result image is computed as follows:
    
    initial_img *  + img *  + 
    NOTE: initial_img and img must be the same shape!
    """
    return cv2.addWeighted(initial_img, , img, , )

def find_lanes(image):
    h = image.shape[0]
    w = image.shape[1]
    v = np.array([[ [0, h], [w/4, h * .6], [w * 3/4, h * .6], [w, h] ]], np.int32)
    #v = np.array([[ [0, h], [w * 0.5, h * .6], [w, h] ]], np.int32)

    grey = grayscale(image)
    smooth = gaussian_blur(grey, 11)
    edges = canny(smooth, 30, 50)

    region = region_of_interest(edges, v)
    hodgelines = hough_lines(region, 1, np.pi / 180, 40, 30, 200)

    return weighted_img(hodgelines, image)
